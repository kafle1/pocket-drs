\section{What it does}
A phone records a cricket delivery; user taps pitch corners and stump rectangles; system returns \textbf{OUT / NOT OUT / UMPIRES CALL} with a 3D Hawk-Eye trajectory.

\section{Pipeline (5 stages)}
\begin{tightenum}
\item \textbf{Calibrate} --- \texttt{solvePnP} on 4 pitch corners + 8 stump-quad points; jointly auto-fits FOV (28--86\textdegree) and pitch length.
\item \textbf{Detect} --- MOG2 motion + HSV colour + ROI mask, with Ultralytics YOLO cricket-ball fallback.
\item \textbf{Link} --- RANSAC over constant-acceleration motion model; outputs $(u,v,t)$ image-space track.
\item \textbf{Reconstruct 3D} --- depth from ball size $d = f_x \cdot R_{\text{real}} / r_{\text{px}}$ ($R_{\text{real}}=0.036$\,m); projectile-fit with $g=9.81$\,m/s$^2$ + bounce reflection (Ribnick 2009 linear solve).
\item \textbf{Decide} --- ICC Rule 36: pitched in line + impact in line + hitting stumps; 25\,mm umpire's-call margin.
\end{tightenum}

\section{Key formulas}
\begin{tightitemize}
\item Pinhole: $u = f_x X/Z + c_x$, \; $v = f_y Y/Z + c_y$
\item $f_x$ from FOV: $f_x = (\max(W,H)/2)/\tan(\text{FOV}/2)$
\item Depth from ball: $d_{\text{m}} = f_x \cdot 0.036 / r_{\text{px}}$
\item Projectile: $z(t) = z_0 + v_z t - \tfrac{1}{2} g t^2$ \; ($g=9.81$)
\item Bounce reflection: $v_{z,\text{post}} = -e\, v_{z,\text{pre}}$ \; ($e\approx 0.55$)
\end{tightitemize}

\section{ICC pitch geometry}
\begin{tightitemize}
\item Pitch: 20.12\,m $\times$ 3.05\,m
\item Stumps: 3 stumps, 0.711\,m tall, 0.114\,m apart (centre-to-centre); 0.018\,m thick $\Rightarrow$ outermost edge $\pm 0.132$\,m
\item Bails: 0.012\,m above stumps
\item LBW in-line corridor: outer edge $\pm 0.036$\,m (ball radius)
\end{tightitemize}

\section{Decision rules (Rule 36)}
\begin{tightitemize}
\item Pitched in line of stumps? (inside corridor)
\item Impact in line of stumps? (offside-of-bat exception \emph{not} implemented)
\item Hitting stumps? (path intersects stump volume + 25\,mm band)
\item \textbf{OUT} only if all three pass with margin $> 25$\,mm
\item Margin within 25\,mm $\Rightarrow$ \textbf{UMPIRES CALL}
\item Any check fails $\Rightarrow$ \textbf{NOT OUT}
\end{tightitemize}

\section{Real-video results (2026-05-24)}
\begin{tightitemize}
\item \texttt{test3.mp4} (zoomed phone, FOV 37\textdegree): 64.1\,km/h, decision \textbf{NOT OUT} (predicted ball misses stumps); reproj 16.6\,px, 28 inliers.
\item Real-phone HEVC truncation handled by pipeline truncation guard.
\end{tightitemize}

\section{Synthetic validation (8 scenarios)}
\begin{tightitemize}
\item \textbf{8/8 PASS} at honest monocular bounds: $\pm$55\,km/h speed, $\pm$175\,cm bounce, $\pm$110\,cm impact.
\item Mean errors: speed 26.7\,km/h, bounce 60\,cm, impact 63\,cm.
\item Sweet-spot setup: 115\,km/h, middle corridor, 5.5--6\,m length, full ICC pitch.
\item Pipeline rejects fits beyond bounds --- never emits confident-wrong decisions.
\end{tightitemize}

\section{Hardening}
\begin{tightitemize}
\item Calibration rejected if reproj $> 8$\,px (large-error pose $=$ meaningless physics).
\item 3D fit rejected if RMS $> 1.0$\,m (untrustworthy).
\item VideoReader uses frame-index seeking (phone HEVC sparse keyframes).
\item Truncation guard stops at last good frame (no stale-frame padding).
\item Camera height $> 0$ invariant prevents physically impossible solves.
\end{tightitemize}

\section{Defense Q\&A}
\begin{tightitemize}
\item \textbf{How accurate?} $\sim$6\,cm bounce, $\sim$80\,cm impact, $\pm$25\,km/h speed monocular. Real Hawk-Eye $=$ sub-cm with 6 cameras.
\item \textbf{Why not multi-camera?} Phone-only $=$ affordable for amateur cricket; trade accuracy for accessibility.
\item \textbf{What if calibration is wrong?} Pipeline rejects (reproj $> 8$\,px $\Rightarrow$ error); no confident-wrong outputs.
\end{tightitemize}
